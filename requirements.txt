# ==========================================
# ChaitanyaFinance System Setup Requirements
# ==========================================

# 1. Core Runtime Environment
# ---------------------------
# Node.js engine version 20.x or higher is required.
node >= 20.20.2
npm >= 10.8.2

# 2. Database Engine
# ---------------------------
# SQLite3 local database is utilized. (sqlite3 binaries compiled via npm).
sqlite3 >= 3.0.0

# 3. NPM Package Dependencies
# ---------------------------
# Core server-side frameworks and database driver wrappers.
express == 4.19.2
sqlite3 == 5.1.7

# 4. Optional Tooling for Development
# -----------------------------------
# Utility helper to automatically restart the node process on file updates.
nodemon >= 3.0.0
